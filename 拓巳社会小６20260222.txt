    const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
    let noiseBuffer = null;

    function createNoiseBuffer() {
        const bufferSize = audioCtx.sampleRate;
        const buffer = audioCtx.createBuffer(1, bufferSize, audioCtx.sampleRate);
        const data = buffer.getChannelData(0);
        for (let i = 0; i < bufferSize; i++) data[i] = Math.random() * 2 - 1;
        noiseBuffer = buffer;
    }
    createNoiseBuffer();

    function playSound(type) {
        const gain = audioCtx.createGain(); gain.connect(audioCtx.destination);
        if (type === 'hit') {
            const source = audioCtx.createBufferSource(); source.buffer = noiseBuffer;
            const filter = audioCtx.createBiquadFilter(); filter.type = 'lowpass'; filter.frequency.value = 800;
            gain.gain.setValueAtTime(0.01, audioCtx.currentTime);
            gain.gain.exponentialRampToValueAtTime(1.0, audioCtx.currentTime + 0.01);
            gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + 0.1);
            source.connect(filter); filter.connect(gain); source.start(); source.stop(audioCtx.currentTime + 0.1);
        } else if (type === 'miss') {
            const osc = audioCtx.createOscillator(); osc.type = 'sawtooth'; osc.frequency.value = 90;
            gain.gain.setValueAtTime(0.3, audioCtx.currentTime); gain.gain.linearRampToValueAtTime(0, audioCtx.currentTime + 0.4);
            osc.connect(gain); osc.start(); osc.stop(audioCtx.currentTime + 0.4);
        } else if (type === 'ippon') {
            const osc1 = audioCtx.createOscillator(); osc1.frequency.value = 523;
            gain.gain.setValueAtTime(0.5, audioCtx.currentTime); gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + 2.0);
            osc1.connect(gain); osc1.start(); osc1.stop(audioCtx.currentTime + 2.0);
        }
    }

    let currentRound = 0, currentQ = 0, warnings = 0;

    function showStartRei() {
        const overlay = document.getElementById("overlay"), msg = document.getElementById("rei-message"), 
              btn = document.getElementById("action-btn"), reward = document.getElementById("special-reward");
        overlay.style.display = "flex";
        reward.innerHTML = "";
        msg.innerText = "正面に礼！"; btn.innerText = "はじめ！";
        btn.onclick = () => { if (audioCtx.state === 'suspended') audioCtx.resume(); overlay.style.display = "none"; loadQuestion(); };
    }

    function loadQuestion() {
        const data = tournament[currentRound], qData = data.qs[currentQ];
        document.getElementById("round-label").innerText = data.name;
        document.getElementById("q-progress").innerText = `第 ${currentQ + 1} / ${data.qs.length} 問`;
        document.getElementById("round-display").innerText = `${data.name} 審査`;
        document.getElementById("opponent-display").innerText = `対戦相手：${data.opponent}`;
        document.getElementById("question-text").innerText = qData.q;
        const area = document.getElementById("choices-area");
        area.innerHTML = "";
        qData.c.forEach(choice => {
            const btn = document.createElement("button"); btn.className = "choice-btn"; btn.innerText = choice;
            btn.onclick = () => checkAnswer(choice); area.appendChild(btn);
        });
    }

    function checkAnswer(choice) {
        if (choice === tournament[currentRound].qs[currentQ].a) {
            playSound('hit'); currentQ++;
            if (currentQ >= tournament[currentRound].qs.length) showIppon(); else loadQuestion();
        } else {
            warnings++; playSound('miss');
            if (warnings >= 3) showDisqualification(); else document.getElementById("warning-display").innerText = `注意 ${warnings}`;
        }
    }

    function showIppon() {
        playSound('ippon');
        const overlay = document.getElementById("overlay"), msg = document.getElementById("rei-message"), 
              btn = document.getElementById("action-btn"), reward = document.getElementById("special-reward");
        overlay.style.display = "flex";
        
        if (currentRound < tournament.length - 1) {
            msg.innerHTML = `<span class="ippon">一本！</span><br>審査通過`;
            btn.innerText = "次の試合へ";
            btn.onclick = () => { currentRound++; currentQ = 0; warnings = 0; document.getElementById("warning-display").innerText = ""; showStartRei(); };
        } else {
            msg.innerHTML = `<span class="ippon" style="font-size:3rem; color:var(--karate-yellow)">昇級しました</span><br>黄色帯です。おめでとう！`;
            reward.innerHTML = `<div class="yellow-belt"></div>`;
            btn.innerText = "道場に戻る";
            btn.onclick = () => location.reload();
        }
    }

    function showDisqualification() {
        const overlay = document.getElementById("overlay"), msg = document.getElementById("rei-message"), btn = document.getElementById("action-btn");
        overlay.style.display = "flex";
